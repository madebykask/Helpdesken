use dh_support

declare @formId int
declare @formFieldName nvarchar(250)
declare @id int

set @formId = 26

declare ff cursor for
select Id, FormFieldName  from tblFormField where Form_Id = @formId
open ff
fetch next from ff 
into  @id, @formFieldName
while @@FETCH_STATUS = 0
begin
		
	print 'update tblFormField set SortOrder = 0, Label = '''', GroupLabel = '''' where Id = '+ convert(nvarchar(20),@id) +' --  ' + @formFieldName
  
	fetch next from ff into @Id, @formFieldName
end
close ff
deallocate ff
